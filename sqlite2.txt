import sqlite3
import csv
from datetime import date, datetime, timedelta

# Подключение к базе данных
conn = sqlite3.connect("library.db")
cursor = conn.cursor()

# Создание таблиц
cursor.executescript("""
DROP TABLE IF EXISTS t_book_usage;
DROP TABLE IF EXISTS t_book;
DROP TABLE IF EXISTS t_reader;

CREATE TABLE t_reader (
    id INTEGER PRIMARY KEY,
    FirstName TEXT NOT NULL,
    LustName TEXT NOT NULL,
    PatronymicName TEXT,
    mail TEXT
);

CREATE TABLE t_book (
    id INTEGER PRIMARY KEY,
    title TEXT NOT NULL,
    author TEXT NOT NULL,
    year INTEGER,
    isbn TEXT
);

CREATE TABLE t_book_usage (
    id INTEGER PRIMARY KEY,
    id_book INTEGER NOT NULL,
    id_reader INTEGER NOT NULL,
    issue_date TEXT NOT NULL,
    due_date TEXT NOT NULL,
    return_date TEXT
);
""")

# Функция для загрузки CSV
def load_csv(filename, table_name):
    with open(filename, 'r', encoding='utf-8') as f:
        reader = csv.reader(f)
        headers = next(reader)  # пропускаем заголовки
        
        # Определяем количество столбцов
        placeholders = ','.join(['?' for _ in headers])
        
        for row in reader:
            if row:  # не пустая строка
                cursor.execute(f"INSERT INTO {table_name} VALUES ({placeholders})", row)

# Загружаем данные
print("Загрузка данных...")
load_csv('catalog_books.csv', 't_book')
load_csv('readers2.csv', 't_reader')
load_csv('issues.csv', 't_book_usage')

# Сохраняем изменения
conn.commit()

conn.close()
